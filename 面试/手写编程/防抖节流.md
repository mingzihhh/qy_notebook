## 防抖节流

```
//防抖
function debounce(fn,delay){
    let timer = null
    return function(){
        let args = auguments
        let context = this
        if(timer) clearTimeout(timer)
        timer = setTimeout(() => {
            fn.apply(context,args)
        },delay)
    }
}
```

```
//节流
function throttle(fn, wait) {
    let timer = null
  
    return function() {
        let context = this,
            args = arguments

        if (!timer) {
            timer = setTimeout(() => {
                fn.apply(context, args)
                timer = null
            }, wait)
        }
    }
}
```
